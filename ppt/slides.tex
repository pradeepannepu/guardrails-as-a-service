% Section 2: Architecture
\section{Architecture}

\begin{frame}{Service Overview}
    \begin{block}{Five Microservices}
        \begin{enumerate}
            \item \textbf{API Gateway} (Node.js) - Port 3000
                  \begin{itemize}
                      \item Authentication \& Authorization
                      \item Request routing
                      \item Rate limiting
                  \end{itemize}
            \item \textbf{Evaluation Service} (Node.js) - Port 3001
            \item \textbf{Policy Service} (Node.js) - Port 3002
            \item \textbf{Audit Service} (Node.js) - Port 3003
            \item \textbf{Model Inference Service} (Python/FastAPI) - Port 8000
        \end{enumerate}
    \end{block}
\end{frame}

% Section 6: Testing
\section{Testing}

\begin{frame}[fragile]{Load Testing with k6}
    \textbf{Prerequisites:}
    \begin{lstlisting}[language=bash]
# macOS
brew install k6

# Windows
choco install k6
\end{lstlisting}

    \textbf{Running Load Tests:}
    \begin{lstlisting}[language=bash]
cd load_test
k6 run load_test.js

# With web dashboard
k6 run --out web-dashboard load_test.js
\end{lstlisting}
\end{frame}

\begin{frame}{Load Test Configuration}
    \begin{block}{Default Load Test Pattern}
        \begin{itemize}
            \item 30s: Ramp up to 10 users
            \item 1m: Ramp up to 25 users
            \item 1m: Ramp up to 50 users
            \item 2m: Maintain 50 users (peak load)
            \item 30s: Ramp down to 0 users
        \end{itemize}
    \end{block}

    \begin{block}{Configuration Notes}
        \begin{itemize}
            \item Update host link to API Gateway URL
            \item Include authentication tokens
            \item Ensure all services are running before testing
        \end{itemize}
    \end{block}
\end{frame}
